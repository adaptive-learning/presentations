    \frametitle{Response Times}

    \begin{itemize}
        \item classic response:
            \begin{itemize}
                \item $r = 0$ - wrong answer
                \item $r = 1$ - correct answer
            \end{itemize}
        \item use of response time:
            \begin{itemize}
                \item $r = 0$ - wrong answer
                \item $r \in [0, 1]$ - correct answer
            \end{itemize}
    \end{itemize}
